import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import com.umfrancisco.Alien;
import com.umfrancisco.Computer;
import com.umfrancisco.Desktop;

@Configuration
public class AppConfig {
	
	@Bean
	public Alien alien(@Autowired Computer computer) {
		Alien obj = new Alien();
		obj.setAge(25);
		obj.setComputer(computer);
		return obj;
	}
	
	@Bean
	public Desktop desktop() {
		return new Desktop();
	}
}

import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import com.umfrancisco.Alien;
import com.umfrancisco.Computer;
import com.umfrancisco.Desktop;
import com.umfrancisco.Laptop;

@Configuration
public class AppConfig {
	
	@Bean
	public Alien alien(@Qualifier("laptop") Computer computer) {
		Alien obj = new Alien(25, computer);
		return obj;
	}
	
	@Bean
	public Desktop desktop() {
		return new Desktop();
	}
	
	@Bean
	public Laptop laptop() {
		return new Laptop();
	}
}

	private Computer computer;

		this.age = age;
		this.computer = computer;
	}

		System.out.println("Coding");

	}
}
