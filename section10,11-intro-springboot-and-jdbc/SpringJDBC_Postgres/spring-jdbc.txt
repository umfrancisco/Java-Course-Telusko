-- schema.sql --
create table student (
	rollno int primary key,
	name varchar(50),
	marks int
);

-- data.sql -- 
insert into student (rollno, name, marks) values 
(101, 'Kiran', 79),
(102, 'Harsh', 68),
(103, 'Sushil', 82);
update student set marks=75 where rollno = 102;
